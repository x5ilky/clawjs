    fn add(this: &Self, right: Right) Output;
}

struct sprite {
    name: string,
    costumes: array<costume>,
}

data vec2 {
    x: int,
    y: int,
}

impl sprite {
    fn new(name: string) sprite {
        return sprite::{
            name: name,
            costumes: array.new()
        }
    }

    fn add_costume(this: &sprite, costume: Costume) Costume {
        this.costumes.push(costume)
        return costume
    }
}
